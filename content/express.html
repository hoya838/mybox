import express from 'express';
const app = express();
app.get('/todos', (req, res) => {   // GET /todos
  res.json([{ title: '공부', done: false }]);
});
app.listen(3000);
만드는 것: FastAPI와 같은 API를 JS로. 비교하며 두 언어 체화.

핵심 개념

1. 라우트 — URL과 핸들러

import express from 'express';
const app = express();
app.use(express.json());        // JSON 본문 파싱

app.get('/todos', (req, res) => {
  res.json([{ title: '공부' }]);
});
app.post('/todos', (req, res) => {
  const todo = req.body;        // 받은 데이터
  res.status(201).json(todo);
});
app.listen(3000);
쉽게: app.get('/주소', 함수) = FastAPI @app.get과 같은 역할. req=요청, res=응답.

2. 미들웨어 — 중간 처리

app.use((req, res, next) => {
  console.log(req.method, req.url);  // 모든 요청 로그
  next();    // 다음 단계로 넘김
});
쉽게: 요청이 거쳐 가는 검문소. 로그·인증·검증을 중간에. next()=통과.

3. DB 연동 + 라우터 분리

// PostgreSQL 연결 (pg 또는 Prisma)
import pg from 'pg';
const db = new pg.Pool({ connectionString: process.env.DB_URL });
const { rows } = await db.query('SELECT * FROM todos');
쉽게: pg로 PostgreSQL에 SQL 날림. 커지면 라우트를 파일별로 분리(MVC).
